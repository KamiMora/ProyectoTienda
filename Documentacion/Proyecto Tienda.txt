    public void eliminarProducto()
    {
        int id = Integer.parseInt(JOptionPane.showInputDialog(null, "Ingrese código del producto", "Producto eliminado", JOptionPane.INFORMATION_MESSAGE)); //(estructura se encuentra en el API)
        String tipo = (JOptionPane.showInputDialog(null, "Ingrese tipo del producto", "Producto eliminado", JOptionPane.INFORMATION_MESSAGE));
        String nombre = (JOptionPane.showInputDialog(null, "Ingrese nombre del producto", "Producto eliminado", JOptionPane.INFORMATION_MESSAGE));
        String marca = (JOptionPane.showInputDialog(null, "Ingrese marca del producto", "Producto eliminado", JOptionPane.INFORMATION_MESSAGE));
        String presentacion = (JOptionPane.showInputDialog(null, "Ingrese presentación del producto", "Producto eliminado", JOptionPane.INFORMATION_MESSAGE));
        int cantidad = Integer.parseInt(JOptionPane.showInputDialog(null, "Ingrese cantidad del producto", "Producto eliminado", JOptionPane.INFORMATION_MESSAGE));
        int precio = Integer.parseInt(JOptionPane.showInputDialog(null, "Ingrese precio del producto", "Producto eliminado", JOptionPane.INFORMATION_MESSAGE));
        
        Producto p = new Producto(id, tipo, nombre, marca, presentacion, cantidad, precio);
        this.bodega.eliminarProducto(p);
        JOptionPane.showMessageDialog(null, "Producto eliminado de la bodega", "Producto eliminado", JOptionPane.INFORMATION_MESSAGE);
    }
